ЧТО ТАКОЕ C#? C# (читается «Си Шарп») — это объектно-ориентированный язык программирования от Microsoft. 🎮 Где используется? • Игры (Unity) • Приложения для Windows • Веб-сайты • Мобильные приложения Запомни: C# — это НЕ скриптовый язык, а полноценный язык для серьезных проектов! ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 2. ТОЧКА ВХОДА — МЕТОД MAIN() Любая программа на C# начинается с метода Main(). 
static void Main()
{
    // Здесь начинается твоя программа
}
⚠️ Важно! C# чувствителен к регистру: • Main() — ✅ правильно • main() — ❌ ошибка! ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 3. КОММЕНТАРИИ В КОДЕ Комментарии — это текст, который игнорируется компилятором. Они нужны, чтобы объяснять код. Однострочный комментарий: 
// Это комментарий до конца строки
Многострочный комментарий: 
/*
   Это многострочный
   комментарий
*/
💡 Совет: Всегда комментируй сложные моменты в коде — через неделю ты сам забудешь, что хотел сделать! ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 4. КЛАССЫ И МЕТОДЫ (ПРОСТЫМ ЯЗЫКОМ) 🏠 Класс — это «чертеж» или «форма» для создания объектов. 📦 Пример из жизни: • Класс = форма для выпечки печенья • Объект = само печенье, испеченное по форме ⚡ Метод — это «действие», которое умеет выполнять класс. 📖 Пример из жизни: • Метод = рецепт «приготовить печенье» • Вызов метода = «давай испечем печенье прямо сейчас!» 
class Program      // Это класс
{
    static void Main()  // Это метод
    {
        // тело метода
    }
}
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 5. ЧТО ДЕЛАЕТ КОМПИЛЯТОР? Компилятор — это переводчик с человеческого языка на язык компьютера. 📝 Твой код (.cs) ↓ ⚙️ КОМПИЛЯТОР (csc.exe) ↓ 💻 Машинный код (.exe) Простыми словами: ты пишешь рецепт на русском, а компилятор переводит его на язык, который понимает повар-компьютер. ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 6. ПОДКЛЮЧЕНИЕ БИБЛИОТЕКИ USING SYSTEM 
using System; — подключает стандартную библиотеку с базовыми классами.
Без using System: 
System.Console.WriteLine("Hello");
С using System: 
Console.WriteLine("Hello"); // короче и удобнее
💡 Запоминалка: using = «дай доступ к коробке с инструментами» ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 📝 ПРИМЕР ПРОГРАММЫ (ВСЕ ВМЕСТЕ): 
using System;      // 1. Подключаем библиотеку

class Program       // 2. Создаем класс
{
    static void Main()  // 3. Точка входа
    {
        // 4. Выводим текст в консоль
        Console.WriteLine("Hello, World!");
    }
}
