ЧТО ТАКОЕ КЛАСС? Класс — это шаблон (чертёж), по которому создаются объекты. 📌 Пример: 
class Car
{
    public string name;
    public int speed;
}
💡 Класс описывает: • какие данные есть • какие действия можно выполнять ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 2. ЧТО ТАКОЕ ОБЪЕКТ? Объект — это экземпляр класса. 
Car car = new Car();
📌 new — создает объект 💡 Аналогия: • класс = чертёж машины • объект = реальная машина ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 3. ПОЛЯ И МЕТОДЫ Класс может содержать: Поля (данные): 
public string name;
Методы (действия): 
public void Drive()
{
    Console.WriteLine("Машина едет");
}
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 4. РАБОТА С ОБЪЕКТОМ 
Car car = new Car();

car.name = "BMW";
car.speed = 200;

Console.WriteLine(car.name);
📌 Доступ к полям через точку (.) ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 5. МОДИФИКАТОРЫ ДОСТУПА • public — доступно везде • private — доступно только внутри класса 
private int secret;
💡 Это нужно для защиты данных ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 6. ОСНОВЫ ООП ООП — это подход, где программа строится на классах и объектах. 📌 Основные идеи: • инкапсуляция (скрытие данных) • повторное использование кода • удобная структура программы ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 7. ПРИМЕР ПОЛНОЙ ПРОГРАММЫ 
class Person
{
    public string name;

    public void SayHello()
    {
        Console.WriteLine("Привет, я " + name);
    }
}

class Program
{
    static void Main()
    {
        Person p = new Person();
        p.name = "Ivan";
        p.SayHello();
    }
}
