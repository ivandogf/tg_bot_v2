ЧТО ТАКОЕ УСЛОВИЕ? Условие позволяет программе принимать решения. 📌 Пример из жизни: • Если идёт дождь → берём зонт • Если нет → идём без зонта В коде это выглядит так же! ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 2. ОПЕРАТОР IF Оператор if проверяет условие. 
int a = 10;

if (a > 5)
{
    Console.WriteLine("a больше 5");
}
💡 Если условие true — код выполнится. ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 3. ОПЕРАТОР ELSE Оператор else выполняется, если условие ложное. 
int a = 3;

if (a > 5)
{
    Console.WriteLine("Да");
}
else
{
    Console.WriteLine("Нет");
}
📌 Здесь выведется "Нет", потому что 3 меньше 5. ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 4. КАК ЭТО РАБОТАЕТ? Программа: 1. Проверяет условие 2. Если true → выполняет if 3. Если false → выполняет else ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 5. ПРИМЕР ИЗ ЖИЗНИ 
int age = 18;

if (age >= 18)
{
    Console.WriteLine("Доступ разрешен");
}
else
{
    Console.WriteLine("Доступ запрещен");
}
