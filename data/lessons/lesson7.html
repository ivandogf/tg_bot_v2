📚 ЗА ЭТОТ УРОВЕНЬ ТЫ ИЗУЧИШЬ: • 📦 Что такое массив • 🔢 Как работает индекс • 📏 Как узнать длину массива • 🔁 Как перебирать массив • 🧠 Где используются массивы ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 1. ЧТО ТАКОЕ МАССИВ? Массив — это набор данных одного типа, объединённых в одну переменную. 📌 Пример: 
int[] numbers = {1, 2, 3, 4};
💡 Вместо создания 4 переменных мы используем один массив. ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 2. ИНДЕКСЫ Каждый элемент массива имеет индекс (позицию). ⚠️ Важно: • индексация начинается с 0 
int[] arr = {10, 20, 30};

arr[0] // 10  
arr[1] // 20  
arr[2] // 30
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 3. ДЛИНА МАССИВА Чтобы узнать количество элементов, используется Length: 
int[] arr = {1, 2, 3};

Console.WriteLine(arr.Length); // 3
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 4. ПЕРЕБОР МАССИВА Можно пройти по всем элементам массива: Через for: 
for (int i = 0; i < arr.Length; i++)
{
    Console.WriteLine(arr[i]);
}
Через foreach: 
foreach (int x in arr)
{
    Console.WriteLine(x);
}
💡 foreach проще и чаще используется. ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 5. ИЗМЕНЕНИЕ ЭЛЕМЕНТОВ Можно менять значения массива: 
int[] arr = {1, 2, 3};

arr[0] = 10;  // теперь массив: 10, 2, 3
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 6. ГДЕ ИСПОЛЬЗУЮТСЯ МАССИВЫ? Массивы применяются: • для хранения списков данных • для обработки результатов • в играх, приложениях, сайтах ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🧠 КРАТКО: • массив = набор значений • индекс начинается с 0 • Length — длина массива • можно перебирать (for / foreach) • можно изменять элементы ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ ✅ ГОТОВ ПРОВЕРИТЬ СЕБЯ?
