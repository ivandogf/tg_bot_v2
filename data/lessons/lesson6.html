ЧТО ТАКОЕ МЕТОД? Метод — это именованный блок кода, который выполняет определённую задачу. 📌 Зачем нужны методы: • чтобы не повторять код • чтобы разбивать программу на части • чтобы делать код понятнее 
void SayHello()
{
    Console.WriteLine("Hello");
}
💡 Метод — это как команда: "сделай что-то". ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 2. ВЫЗОВ МЕТОДА Метод не выполнится сам — его нужно вызвать: 
SayHello();
📌 После вызова выполнится код внутри метода. ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 3. МЕТОДЫ С ПАРАМЕТРАМИ Методы могут принимать данные (параметры): 
void PrintNumber(int num)
{
    Console.WriteLine(num);
}
Вызов: 
PrintNumber(5);
💡 Параметры — это входные данные для метода. ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 4. RETURN (ВОЗВРАТ ЗНАЧЕНИЯ) Метод может вернуть результат: 
int Sum(int a, int b)
{
    return a + b;
}
Использование: 
int result = Sum(2, 3);
📌 result будет равен 5 ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 5. VOID И ВОЗВРАЩАЕМЫЕ МЕТОДЫ • void — метод ничего не возвращает • int, string и др. — метод возвращает значение 
void SayHi()
{
    Console.WriteLine("Hi");
}

int GetNumber()
{
    return 10;
}
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 6. ПРИМЕР В РЕАЛЬНОЙ ПРОГРАММЕ 
int Square(int x)
{
    return x * x;
}

Console.WriteLine(Square(4)); // 16
