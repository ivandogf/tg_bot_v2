АРИФМЕТИЧЕСКИЕ ОПЕРАТОРЫ Операторы используются для выполнения математических действий. Основные операторы: • + — сложение • - — вычитание • * — умножение • / — деление • % — остаток от деления 
int a = 5 + 3;   // 8
int b = 10 % 3;  // 1
💡 Запомни: % показывает остаток, а не результат деления! ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 2. ОПЕРАТОРЫ СРАВНЕНИЯ Они используются для проверки условий. Основные операторы: • == — равно • != — не равно • < — меньше • > — больше 
int a = 5;
int b = 3;

if (a > b)
{
    Console.WriteLine("a больше b");
}
⚠️ Важно! • = — это присваивание • == — это сравнение Не путай их! ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 3. ИНКРЕМЕНТ (++) Инкремент увеличивает значение переменной на 1. 
int x = 5;
x++;   // теперь x = 6
Также есть уменьшение: • -- уменьшает значение на 1 
x--;   // теперь x = 5
💡 Запоминалка: ++ → плюс один -- → минус один ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ 🔹 4. ПРИМЕР ИСПОЛЬЗОВАНИЯ 
int a = 10;
int b = 4;

Console.WriteLine(a + b);  // 14
Console.WriteLine(a % b);  // 2

if (a > b) //  if это условный оператор об этом  мы поговорим в теории на следующих уровнях
{
    Console.WriteLine("a больше b");
}
